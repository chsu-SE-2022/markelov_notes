```cpp
template<class T>
class vector_pointer : private vector {
public:
	vector_pointer();
	void print() const {
		vector::print();
	}
	void input(T* d) {
		vector::input(d);
	}
	T* find(T* d) const {
		return reinterpret_cast<T*>(vector::find(d));
	}
};

vector_pointer<char*> v1, v2, v3;
vector_pointer<type*> t1, t2, t3;
```
