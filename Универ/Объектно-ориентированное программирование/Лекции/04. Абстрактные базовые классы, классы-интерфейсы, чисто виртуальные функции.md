## Перегрузка виртуальных функций
```cpp
class B {
public:
virtual void f(int); 
virtual void f(float);  //перегруженные функции
};
class D: public B {
public:
void f(int);  //замещает B::f(int), скрывает B::f(float)
}
D d;
B b;
*pb = &d;
b.f(9);
b.f(9.5);
d.f(9);  //D::f(int)
d.f(9.5);  //B::f(float)
pb->f(9);  //D::f(int) - замещает B::f(int)
pb->f(9.5);  //B::f(float) - наследование
```
